== *Container Objects*

[source,java]
----
public class Service {

   @CubeDescriptor
   public static Archive<?> service() {
      return ShrinkWrap.create(GenericArchive.class)
         .add("my_file.txt")
         .add(Descriptors.create(CubeDescriptor.class)
            .build(Descriptors.create(DockerDescriptor.class)
                  .from("tomcat")
                  .add("my_file.txt")
            )
         );
   }

   @Cube("mysql_1")
   private Backend backend;

   @ArquillianResource
   private HostIP host;

   public URL getServiceURL() {
      return new URL(host.getIP(), host.getPort());
   }
}
----

[.topic.source]
== *Container Objects*

[source,java]
----
@RunWith(Arquillian.class)
public class Test {

   @Cube("service1")
   private Service service;

   @Test
   public void shouldX() {
      given().
         baseUri(service.getServiceURL()).
      when().
         get("/service/beer").
      then().
         assertThat()
            .body("name", is(equalTo("Snake Venom")))
            .body("abv", is(equalTo("67.5")));
   }
}
----
